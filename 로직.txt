def process_single_file(filename, input_dir, output_dir, use_psr, sm, cm):
    input_path = os.path.join(input_dir, filename)
    is_ext = "is_exception" in filename 
    
    with open(input_path, "r", encoding="utf-8") as f: 
        text = f.read()
    
    # 1. 변환 수행
    if is_ext:
        final = transform(text, sm, cm, psr=use_psr)
        out_filename = os.path.splitext(filename)[0] + ".exc" 
    else:
        final = transform(preprocess(text), sm, cm, psr=use_psr)
        out_filename = os.path.splitext(filename)[0] + ".pat"
        
    # 💡 [핵심 버그 픽스] EOL/EOF 유령 에러 원천 차단 로직
    # 1. splitlines()로 쪼갠 뒤 rstrip()으로 모든 줄의 우측 숨은 공백/탭을 완벽히 제거합니다.
    # 2. \n으로 다시 조립한 뒤, 파일 맨 마지막에 반드시 엔터( \n )를 하나 강제로 추가합니다.
    final = "\n".join(line.rstrip() for line in final.splitlines()) + "\n"
    
    # 2. 파일 저장
    out_path = os.path.join(output_dir, out_filename)
    with open(out_path, "w", encoding="utf-8") as f: 
        f.write(final)
        
    # 3. 권한 및 결과 반환
    if is_ext:
        os.chmod(out_path, 0o777)
        return f"  [SPECIAL] {filename}  ->  {out_path} (구조 유지 완벽 / 777 권한 적용)"
    else:
        return f"  [OK] {filename}  ->  {out_path}"
